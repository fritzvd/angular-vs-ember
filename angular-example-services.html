angular.module('example', []); // initiate module and declare dependencies: none;

angular.module('example')
  .controller('BasicCtrl', ['$scope', 'Economics', function ($scope, Economics) {
    $scope.whisky = {
      price: 40,
      name: "Lagavullin"
    };

    $scope.inflation = function () {
      $scope.whisky.price = Economics.inflation($scope.whisky.price);
    };
  }]);

angular.module('example')
  .service('Economics', function () {

    var _rate = 1.02; // 2%
    
    function Inflation (price) {
      return price * _rate;
    };

    return {
      inflation: Inflation  
    }
  });



<section ng-controller="BasicCtrl">
  <h5> \{\{ whisky.name \}\} \{\{ whisky.price | number: 2 \}\} </h5>
  <input type="range" ng-model="whisky.price"></input>
  <input type="text" ng-model="whisky.name"></input>
  <button ng-click="inflation()"></button>
</section>
